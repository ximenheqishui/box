打包所有组件到一个js和css中，可以给其他非ts的项目使用

测试分页组件

基本使用方式 
                let pagination1 = $$.pagination({
                    elem: '.page-content',
                    currentPage: 1,
                    pageSize: 10,
                    total: 0,
                    sizeChange: onSizeChange,
                    currentChange: onCurrentChange
                })
                pagination1.setTotal(1000)
                pagination1.setCurrentPage(1)
                // 每页多少变化
                function onSizeChange(csize,cpn) {
                    console.log('分页大小变化',cpn)
                }
                // 页码变化
                function onCurrentChange (cpn) {
                    console.log('页码发生变化',cpn)
                }
