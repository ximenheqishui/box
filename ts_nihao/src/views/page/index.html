开发测试分页组件

基本使用方式 
                let pagination1 = pagination({
                    elem: '.page-content',
                    currentPage: 1,
                    pageSize: 10,
                    total: 1000,
                    sizeChange: onSizeChange,
                    currentChange: onCurrentChange
                })
修改分页个数 
                    let pagination1 = pagination({
                        elem: '.page-content2',
                        pagerCount: 12,
                        currentPage: 1,
                        pageSize: 10,
                        total: 1000,
                        sizeChange: onSizeChange,
                        currentChange: onCurrentChange
                    })
